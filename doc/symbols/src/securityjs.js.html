 93 /**
 94  * Este método se encarga de verificar que todos los permisos definidos
 95  * en `require` se encuentren en `permisos`.
 96  * @function
 97  *
 98  * @public
 99  * @author <a href="mailto:mxbg.py@gmail.com">Maximiliano Báez</a>
100  * @name SecurityJS#contains
101  * @param require {String[]} El array de permisos necesarios
102  * @param data {String[]} El array de permisos obtenidos.
103  * @return {Boolean} True si todos existen todos los permisos,
104  *          en caso contrario retorna False.
105  */
106 _securityjs.contains = function(require, data){
107     for(var i=0; i<require.length; i++){
108         var permiso = require[i].trim();
109         var status = (data.indexOf(permiso) >= 0);
110         if(!status){
111             return false;
112         }
113     }
114     return true;
115 }

117 /**
118  * Este método se encarga de verificar los permisos del usuario y
119  * realizar una opreacion dependiendo del modo asociado al tag HTML.
120  * @function
121  *
122  * @public
123  * @author <a href="mailto:mxbg.py@gmail.com">Maximiliano Báez</a>
124  * @name SecurityJS#initialize
125  */
126 _securityjs.initialize = function (){
127     //se obtienen todos los permisos
128     var thiz = this;
129     //se obtienen todos los elementos que tienen el atributo require
130     $(this.scope+' [data-require]').each(function(){
131         // se obtienen los permisos definidos en atributo require del
132         // componente de html.
133         var toSplit = $(this).attr('data-require');
134         //se obtiene el modo de procesar el tag
135         var mode = $(this).attr('data-require-fail');
136         //por default se toma remove
137         if(typeof mode == "undefined"){
138             mode="remove";
139         }
140         //se separa los elementos
141         var require = toSplit.split(';');
142         //se verifica si existen posee los permisos.
143         if(!thiz.contains(require, thiz.data)){
144             //se verifica si el modo esta soportado
145             if(thiz[mode] instanceof Function){
146                 //se invoca al encargado de procesar el elemento
147                 thiz[mode]({
148                     el :$(this),
149                     require : require
150                 });
151             }else{
152                 throw  'Value of data-require-fail `'+mode+'` is not valid';
153             }
154             //se invoca al handler si es que este existe
155             if(thiz.handler instanceof Function){
156                 options.handler({
157                     el :$(this),
158                     require : require
159                 });
160             }
161         }
162         thiz.afterProcess($(this))
163     });
164 };

166 /**

189  *          dom asociado al permiso.
190  *
191  * @config [hide]{Function}  Se encarga de manejar aquellos elementos que
192  *          cuenten con el valor `hide` en el atributo `data-require-fail`.
193  *          Por default añade el style `visibility:hidden` para ocultar el
194  *          dom asociado al permiso.
195  *
196  * @config [error]{Function} Se encarga de manejar aquellos elementos que

199  *
200  * @config [afterProcess]{Function} Este método se invoca una vez que el
201  *         `el` fue procesado. Por default se eliminan los atributos
202  *         `data-require` y `data-require-fail`.
203  */
204 var SecurityJS = function (options) {
205     if(!options){options={}}
206     //se define la clase
207     var clazz={};
208     //se realiza un extend de json principal para y un merge entre
209     //los atributos y metodos.
210     $.extend(true,clazz,_securityjs, options);
211     //se invoca al constructor de la clase
212     clazz.initialize(options);
213     //se retorna la definicion de la clase
214     return clazz;
215 }
